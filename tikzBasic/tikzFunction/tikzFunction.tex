            \draw (0,\y)--(-.1,\y);
        }

\begin{figure}[htp]
    \centering
    \begin{tikzpicture}[>=latex,scale=1.4]   
        \draw[->] (-1,0) -- (7,0)node[right]{$x$};
        \draw[->] (0,-2) -- (0,2)node[right]{$y$};
        \draw[domain=-pi/6:13*pi/6,samples=1000,dashed]
            plot(\x,{sin(\x r)});
        \draw[domain=0:2*pi,samples=1000,very thick]
            plot(\x,{sin(\x r)});
        \draw (0,1)node[left]{1}--(.1,1);
        \draw (0,1.5)node[left]{$\frac{3}{2}$}--(.1,1.5);
        \draw (0,-1)--(.1,-1)node[right]{-1};
        \draw (0,-1.5)--(.1,-1.5)node[right]{-$\frac{3}{2}$};
        \foreach \x/\xtext in {-1/-\frac{\pi}{6},1/\frac{\pi}{6},3/\frac{\pi}{2},5/\frac{5\pi}{6},8/\frac{4\pi}{3},10/\frac{5\pi}{3},12/2\pi,13/\frac{13\pi}{6}}  
        {
            \draw (\x*pi/6,0)node[below]{$\xtext$} -- (\x*pi/6,0.1);
        }
        \foreach \x/\xtext in {2/\frac{\pi}{3},4/\frac{2\pi}{3},6/\pi,7/\frac{7\pi}{6},9/\frac{3\pi}{2},11/\frac{11\pi}{6}}  
        {
            \draw (\x*pi/6,0)node[above]{$\xtext$} -- (\x*pi/6,0.1);
        }
        \node at (.2,-.2){$O$};
        \foreach \x in {-1,0,1,...,13}
        {
            \draw (\x*pi/6,{sin(\x * pi/6 r)})[fill=black] circle (1pt);
        }
    \end{tikzpicture}
    \caption{三角函数}
\end{figure}
